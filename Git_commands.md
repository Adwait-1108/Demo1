# GIT COMMANDS





git --version 								##Command to see if git is installed in the system.

git config --global user.name "Adwait" 					##Command to set the username of the current device.

git config --global user.email "adwaitvid@gmail.com"			##Command to set the the email associated with the current device.

git config --list 							##Command to see the current username and email.

git init								##Command to create a invisible get folder to convert the current folder into a 									  repository.

git add app.py								##Command to add the file individually to the staging area for appending and tracking.

git commit -m "app.py added with one function even\_odd"			##Command to commit the dile so that everyone can see and track the file.

git status								##Command to see the current status of the folder.

git log									##Command to see the logs of the folder.

git log --oneline							##Command to see the logs in one line.

git diff 227eb44 80c6d23						##Command to check the difference between the file logs.

git rm filename 							##Command to delete a file.

git remote add origin https://github.com/Adwait-1108/Demo1.git		##Command to link the GitHub repository with local repository.

git branch -M main							##Command to change the branch from master to main.

git push -u origin main							##Command to push the file from local to remote mode.

git restore --staged new.txt						##Command to restore a file which was added by mistake.

git reset new.txt							##Command to reset a file which was committed by mistake.
